#Music_overtunes
Generating sound with Karplus String Algorithm

Inputs:
  --display:
      plays the generatet sounds
  --play:
      play randon or from file
  --piano:
      can play the sound from your keybord
  --japan:
      use japanese scale
  --china:
      use chinese scale
  --inFile:
      plays from input file
  --outFile:
      save the piano keys to file
